let editingCategoryId = null; let adminCategories = []; function resetCategoryForm() { editingCategoryId = null; document.getElementById("category-name").value = ""; document.getElementById("category-slug").value = ""; document.getElementById("category-sort-order").value = ""; document.getElementById("category-is-active").checked = true; } function getCategoryFormData() { return { name: document.getElementById("category-name").value.trim(), slug: document.getElementById("category-slug").value.trim().toLowerCase(), sort_order: document.getElementById("category-sort-order").value, is_active: document.getElementById("category-is-active").checked }; } async function loadAdminCategories() { try { const res = await adminApiFetch("/admin/categories"); const rows = await res.json(); adminCategories = rows || []; const body = document.getElementById("category-table-body"); const select = document.getElementById("product-category"); // 更新商品分類下拉 if (select) { select.innerHTML = adminCategories .filter(row => row.is_active) .map(row => ``) .join(""); } if (!adminCategories.length) { body.innerHTML = `目前沒有分類`; return; } body.innerHTML = adminCategories.map(row => ` ${row.name} ${row.slug} ${row.sort_order ?? 0} ${row.is_active ? "啟用" : "停用"} `).join(""); } catch (err) { console.error("載入分類失敗", err); alert("載入分類失敗"); } } function editCategory(row) { editingCategoryId = row.id; document.getElementById("category-name").value = row.name || ""; document.getElementById("category-slug").value = row.slug || ""; document.getElementById("category-sort-order").value = row.sort_order ?? ""; document.getElementById("category-is-active").checked = !!row.is_active; } async function saveCategory() { try { const payload = getCategoryFormData(); const url = editingCategoryId ? `/admin/categories/${editingCategoryId}` : "/admin/categories"; const method = editingCategoryId ? "PUT" : "POST"; const res = await adminApiFetch(url, { method, body: JSON.stringify(payload) }); const data = await res.json(); if (!res.ok) { alert(data.error || "儲存失敗"); return; } alert(editingCategoryId ? "分類已更新" : "分類已新增"); resetCategoryForm(); loadAdminCategories(); } catch (err) { console.error("儲存分類失敗", err); alert("儲存分類失敗"); } } async function deleteCategory(id) { if (!confirm("確定刪除分類？")) return; try { const res = await adminApiFetch(`/admin/categories/${id}`, { method: "DELETE" }); const data = await res.json(); if (!res.ok) { alert(data.error || "刪除失敗"); return; } loadAdminCategories(); } catch (err) { console.error("刪除分類失敗", err); alert("刪除分類失敗"); } }
